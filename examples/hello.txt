fn max(a: i32, b: i32) -> i32 {
    let x: i32 = 0;
    if (a > b) {
        x = a;
    } else {
        return b;
    }
    return x;
}

fn main() -> i32 {
    let x: i32 = 0;
    x = max(3, 4);
    return x + square(2);
}

fn square(x: i32) -> i32 {
    return x * x;
}

fn no_return() -> void {
    return;
}
