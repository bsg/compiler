fn max(a: i32, b: i32) -> i32 {
    if (a > b) {
        return a;
    } else {
        return b;
    }
}

fn main() -> i32 {
    let x: i32 = 0;
    x = max(3, 4);

    if (x > 1) {
        if (x < 5) {
            return x;
        } else {
            return 0;
        }
    } else {
        return 1;
    }
}

fn square(x: i32) -> i32 {
    return x * x;
}
